from security import hash_password, verify_password


def test_long_password():
    # contraseÃ±a > 72 bytes
    long_pw = 'p' * 200
    print(f"Long password length: {len(long_pw)}")
    h = hash_password(long_pw)
    print(f"Hash generated (len={len(h)}): {h}")
    ok = verify_password(long_pw, h)
    print(f"Verification result: {ok}")


if __name__ == '__main__':
    test_long_password()
